#### 3.2.1 混合推荐算法设计

```java
/**
 * 混合推荐算法
 * 权重分配：
 * - 协同过滤：40%（用户行为相似度）
 * - 内容推荐：30%（课程特征匹配）
 * - 知识推荐：20%（学习路径连贯性）
 * - 热门推荐：10%（全局热门）
 */
@Service
public class HybridRecommendationEngine {
    
    private static final double WEIGHT_COLLABORATIVE = 0.40;
    private static final double WEIGHT_CONTENT = 0.30;
    private static final double WEIGHT_KNOWLEDGE = 0.20;
    private static final double WEIGHT_TRENDING = 0.10;
    
    @Autowired
    private CollaborativeFilteringStrategy collaborativeStrategy;
    
    @Autowired
    private ContentBasedStrategy contentStrategy;
    
    @Autowired
    private KnowledgeBasedStrategy knowledgeStrategy;
    
    @Autowired
    private TrendingStrategy trendingStrategy;
    
    public List<RecommendationResult> recommend(Long userId, int limit) {
        // 1. 并行执行各策略
        CompletableFuture<List<RecommendationResult>> collaborativeFuture = 
            CompletableFuture.supplyAsync(() -> 
                collaborativeStrategy.recommend(userId, limit * 2));
        
        CompletableFuture<List<RecommendationResult>> contentFuture = 
            CompletableFuture.supplyAsync(() -> 
                contentStrategy.recommend(userId, limit * 2));
        
        CompletableFuture<List<RecommendationResult>> knowledgeFuture = 
            CompletableFuture.supplyAsync(() -> 
                knowledgeStrategy.recommend(userId, limit * 2));
        
        CompletableFuture<List<RecommendationResult>> trendingFuture = 
            CompletableFuture.supplyAsync(() -> 
                trendingStrategy.recommend(userId, limit));
        
        // 2. 等待所有策略完成
        CompletableFuture.allOf(collaborativeFuture, contentFuture, 
                               knowledgeFuture, trendingFuture).join();
        
        // 3. 加权融合
        Map<Long, RecommendationResult> merged = new HashMap<>();
        
        mergeResults(merged, collaborativeFuture.get(), WEIGHT_COLLABORATIVE);
        mergeResults(merged, contentFuture.get(), WEIGHT_CONTENT);
        mergeResults(merged, knowledgeFuture.get(), WEIGHT_KNOWLEDGE);
        mergeResults(merged, trendingFuture.get(), WEIGHT_TRENDING);
        
        // 4. 排序并返回Top N
        return merged.values().stream()
                .sorted(Comparator.comparing(RecommendationResult::getFinalScore).reversed())
                .limit(limit)
                .collect(Collectors.toList());
    }
    
    private void mergeResults(Map<Long, RecommendationResult> merged,
                             List<RecommendationResult> results, double weight) {
        for (RecommendationResult result : results) {
            Long courseId = result.getCourseId();
            if (merged.containsKey(courseId)) {
                // 已存在，累加分数
                RecommendationResult existing = merged.get(courseId);
                existing.addScore(result.getScore() * weight);
                existing.addSource(result.getSource());
            } else {
                // 新加入
                result.setFinalScore(result.getScore() * weight);
                merged.put(courseId, result);
            }
        }
    }
}
```

#### 3.2.2 协同过滤算法

```java
/**
 * 基于用户的协同过滤
 */
@Component
public class CollaborativeFilteringStrategy implements RecommendationStrategy {
    
    @Autowired
    private UserBehaviorMapper userBehaviorMapper;
    
    @Autowired
    private CourseSimilarityMapper similarityMapper;
    
    @Override
    public List<RecommendationResult> recommend(Long userId, int limit) {
        // 1. 获取目标用户的行为向量
        Map<Long, Double> userVector = getUserBehaviorVector(userId);
        
        // 2. 找到相似用户
        List<SimilarUser> similarUsers = findSimilarUsers(userId, userVector, 20);
        
        // 3. 聚合相似用户的偏好
        Map<Long, Double> candidateScores = new HashMap<>();
        for (SimilarUser similarUser : similarUsers) {
            List<UserBehavior> behaviors = userBehaviorMapper
                .findByUserId(similarUser.getUserId());
            
            for (UserBehavior behavior : behaviors) {
                Long courseId = behavior.getCourseId();
                if (!userVector.containsKey(courseId)) { // 过滤已学过的
                    double score = behavior.getBehaviorValue() * similarUser.getSimilarity();
                    candidateScores.merge(courseId, score, Double::sum);
                }
            }
        }
        
        // 4. 转换为推荐结果
        return candidateScores.entrySet().stream()
                .map(entry -> RecommendationResult.builder()
                        .courseId(entry.getKey())
                        .score(entry.getValue())
                        .source("collaborative")
                        .reason("与您相似的用户都在学")
                        .build())
                .sorted(Comparator.comparing(RecommendationResult::getScore).reversed())
                .limit(limit)
                .collect(Collectors.toList());
    }
    
    /**
     * 计算用户相似度（余弦相似度）
     */
    private double calculateSimilarity(Map<Long, Double> vector1, 
                                      Map<Long, Double> vector2) {
        double dotProduct = 0.0;
        double norm1 = 0.0;
        double norm2 = 0.0;
        
        for (Long courseId : vector1.keySet()) {
            double v1 = vector1.get(courseId);
            norm1 += v1 * v1;
            
            if (vector2.containsKey(courseId)) {
                double v2 = vector2.get(courseId);
                dotProduct += v1 * v2;
            }
        }
        
        for (double v : vector2.values()) {
            norm2 += v * v;
        }
        
        return dotProduct / (Math.sqrt(norm1) * Math.sqrt(norm2));
    }
}
```

#### 3.2.3 内容推荐算法

```java
/**
 * 基于内容的推荐
 */
@Component
public class ContentBasedStrategy implements RecommendationStrategy {
    
    @Autowired
    private UserPreferenceMapper preferenceMapper;
    
    @Autowired
    private CourseMapper courseMapper;
    
    @Override
    public List<RecommendationResult> recommend(Long userId, int limit) {
        // 1. 获取用户偏好
        List<UserPreference> preferences = preferenceMapper.findByUserId(userId);
        Map<String, Double> preferenceVector = buildPreferenceVector(preferences);
        
        // 2. 获取用户已学课程
        Set<Long> learnedCourses = getLearnedCourses(userId);
        
        // 3. 计算候选课程匹配度
        List<Course> candidates = courseMapper.findAllActive();
        List<RecommendationResult> results = new ArrayList<>();
        
        for (Course course : candidates) {
            if (learnedCourses.contains(course.getId())) {
                continue;
            }
            
            double score = calculateContentScore(course, preferenceVector);
            if (score > 0.3) { // 阈值过滤
                results.add(RecommendationResult.builder()
                        .courseId(course.getId())
                        .score(score)
                        .source("content")
                        .reason("符合您的" + getTopPreference(preferences) + "偏好")
                        .build());
            }
        }
        
        return results.stream()
                .sorted(Comparator.comparing(RecommendationResult::getScore).reversed())
                .limit(limit)
                .collect(Collectors.toList());
    }
    
    private double calculateContentScore(Course course, Map<String, Double> preferences) {
        double score = 0.0;
        
        // 类别匹配
        if (preferences.containsKey("category:" + course.getCategory())) {
            score += preferences.get("category:" + course.getCategory()) * 0.4;
        }
        
        // 难度匹配
        if (preferences.containsKey("difficulty:" + course.getDifficulty())) {
            score += preferences.get("difficulty:" + course.getDifficulty()) * 0.3;
        }
        
        // 标签匹配
        for (String tag : course.getTags()) {
            if (preferences.containsKey("tag:" + tag)) {
                score += preferences.get("tag:" + tag) * 0.3;
            }
        }
        
        return Math.min(score, 1.0);
    }
}
```

```java
                .userId(userId)
                .courseId(courseId)
                .behaviorType(behaviorType)
                .behaviorValue(BEHAVIOR_WEIGHTS.getOrDefault(behaviorType, 1.0))
                .ipAddress(getClientIp())
                .userAgent(getUserAgent())
                .createdTime(LocalDateTime.now())
                .build();

        kafkaTemplate.send("user-behavior", new UserBehaviorEvent(behavior));
    }
    
    /**
     * 批量处理行为数据（Kafka消费者）

    @KafkaListener(topics = "user-behavior", groupId = "recommendation-group")
```

```java
/**

@Component
public class ColdStartHandler {
    
    @Autowired
    private UserPreferenceMapper preferenceMapper;
    
    @Autowired
    private CourseMapper courseMapper;
    
    /**
     * 新用户推荐
     */
    public List<RecommendationResult> recommendForNewUser(NewUserProfile profile, int limit) {
        // 1. 基于注册信息推荐
        List<RecommendationResult> results = new ArrayList<>();
        
        // 根据兴趣标签推荐
        if (profile.getInterests() != null) {
            for (String interest : profile.getInterests()) {
                List<Course> courses = courseMapper.findByCategory(interest);
                for (Course course : courses) {
                    results.add(RecommendationResult.builder()
                            .courseId(course.getId())
                            .score(0.8)
                            .source("cold_start_interest")
                            .reason("符合您感兴趣的" + interest + "领域")
                            .build());
                }
            }
        }
        
        // 2. 补充热门课程
        List<RecommendationResult> trending = getTrendingCourses(limit / 2);
        results.addAll(trending);
        
        // 3. 补充新手入门课程
        List<RecommendationResult> beginner = getBeginnerCourses(limit / 4);
        results.addAll(beginner);
        
        // 去重并排序
        return results.stream()
                .collect(Collectors.groupingBy(RecommendationResult::getCourseId))
                .values().stream()
                .map(list -> list.get(0))
                .sorted(Comparator.comparing(RecommendationResult::getScore).reversed())
                .limit(limit)
                .collect(Collectors.toList());

    /**

     */
    public void promoteNewCourse(Long courseId) {
        // 基于内容相似度推荐给可能感兴趣的用户
        Course newCourse = courseMapper.findById(courseId);
        
        // 找到相似课程的学习者
        List<Long> similarCourseIds = findSimilarCourses(newCourse);
        Set<Long> targetUsers = new HashSet<>();
        
        for (Long similarId : similarCourseIds) {
            List<Long> learners = userBehaviorMapper.findLearnersByCourse(similarId);
            targetUsers.addAll(learners);
        }
        
        // 生成推荐
        for (Long userId : targetUsers) {
            recommendationMapper.insert(Recommendation.builder()
                    .userId(userId)
                    .courseId(courseId)
                    .recommendationType("cold_start_new_course")
                    .recommendationScore(0.75)
                    .recommendationReason("新课程，与您学过的内容相关")
                    .build());
        }
    }
}
```

```sql
-- 物化视图（热门课程统计）
CREATE TABLE trending_courses_mv (
    course_id BIGINT PRIMARY KEY,
    view_count INT,
    enroll_count INT,
    score DOUBLE,
    updated_time TIMESTAMP
);

-- 定时更新物化视图
CREATE EVENT update_trending_mv
ON SCHEDULE EVERY 1 HOUR
DO
    INSERT INTO trending_courses_mv (course_id, view_count, enroll_count, score, updated_time)
    SELECT 
        course_id,
        COUNT(CASE WHEN behavior_type = 'view' THEN 1 END) as view_count,
        COUNT(CASE WHEN behavior_type = 'enroll' THEN 1 END) as enroll_count,
        AVG(behavior_value) as score,
        NOW()
    FROM user_behaviors
    WHERE created_time > DATE_SUB(NOW(), INTERVAL 7 DAY)
    GROUP BY course_id
    ON DUPLICATE KEY UPDATE
        view_count = VALUES(view_count),
        enroll_count = VALUES(enroll_count),
        score = VALUES(score),
        updated_time = VALUES(updated_time);
```

```java
@Service
public class ABTestService {
    
    /**
     * 分流策略
     */
    public String assignStrategy(Long userId) {

        if (userId % 10 == 0) {

    /**
     * 记录实验数据
     */
    public void logExperiment(Long userId, String strategy, 
                             RecommendationResult recommendation) {
        ExperimentLog log = ExperimentLog.builder()
                .userId(userId)
                .strategy(strategy)
                .recommendationId(recommendation.getCourseId())
                .isClicked(false)
                .isEnrolled(false)
                .build();
        
        experimentLogMapper.insert(log);
    }
    
    /**
     * 计算实验效果
     */
    public ExperimentResult calculateResult(String strategy) {
        List<ExperimentLog> logs = experimentLogMapper.findByStrategy(strategy);
        
        long total = logs.size();
        long clicked = logs.stream().filter(ExperimentLog::getIsClicked).count();
        long enrolled = logs.stream().filter(ExperimentLog::getIsEnrolled).count();
        
        return ExperimentResult.builder()
                .strategy(strategy)
                .ctr((double) clicked / total)
                .cvr((double) enrolled / total)
                .build();
    }
}
```
