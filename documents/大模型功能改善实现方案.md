### 3.2 多模型工厂实现

```java
@Component
public class LanguageModelFactory {
    
    @Autowired
    private ModelConfigs modelConfigs;
    
    private final Map<String, ChatLanguageModel> modelCache = new ConcurrentHashMap<>();
    
    public ChatLanguageModel getModel(String provider) {
        return modelCache.computeIfAbsent(provider, this::createModel);
    }
    
    private ChatLanguageModel createModel(String provider) {
        switch (provider.toLowerCase()) {
            case "wenxin":
                return createWenxinModel();
            case "qwen":
                return createQwenModel();
            case "ollama":
                return createOllamaModel();
            default:
                throw new IllegalArgumentException("Unknown provider: " + provider);
        }
    }
    
    private ChatLanguageModel createWenxinModel() {
        ModelConfig config = modelConfigs.getPrimary();
        return WenxinChatModel.builder()
                .apiKey(config.getApiKey())
                .secretKey(config.getSecretKey())
                .modelName(config.getModelName())
                .temperature(config.getTemperature())
                .maxTokens(config.getMaxTokens())
                .timeout(config.getTimeout())
                .build();
    }
    
    private ChatLanguageModel createQwenModel() {
        ModelConfig config = modelConfigs.getFallback();
        return QwenChatModel.builder()
                .apiKey(config.getApiKey())
                .modelName(config.getModelName())
                .temperature(config.getTemperature())
                .maxTokens(config.getMaxTokens())
                .timeout(config.getTimeout())
                .build();
    }
    
    private ChatLanguageModel createOllamaModel() {
        ModelConfig config = modelConfigs.getLocal();
        return OllamaChatModel.builder()
                .baseUrl(config.getBaseUrl())
                .modelName(config.getModelName())
                .temperature(config.getTemperature())
                .timeout(config.getTimeout())
                .build();
    }
}
```

### 3.3 模型路由与降级服务

```java
@Service
@Slf4j
public class ModelRoutingService {
    
    @Autowired
    private LanguageModelFactory modelFactory;
    
    @Autowired
    private ModelConfigs modelConfigs;
    
    private final CircuitBreakerRegistry circuitBreakerRegistry = 
        CircuitBreakerRegistry.ofDefaults();
    
    /**
     * 带降级的模型调用
     */
    public String generateWithFallback(String prompt) {
        // 尝试主模型
        try {
            CircuitBreaker cb = circuitBreakerRegistry.circuitBreaker("primary");
            return cb.executeSupplier(() -> {
                ChatLanguageModel model = modelFactory.getModel(
                    modelConfigs.getPrimary().getProvider()
                );
                return model.generate(prompt);
            });
        } catch (Exception e) {
            log.warn("Primary model failed, falling back to secondary", e);
        }
        
        // 降级到备用模型
        try {
            ChatLanguageModel fallbackModel = modelFactory.getModel(
                modelConfigs.getFallback().getProvider()
            );
            return fallbackModel.generate(prompt);
        } catch (Exception e) {
            log.error("Fallback model also failed", e);
            throw new AIException("All models failed");
        }
    }
    
    /**
     * 流式生成（SSE）
     */
    public void generateStream(String prompt, Consumer<String> onToken) {
        ChatLanguageModel model = modelFactory.getModel(
            modelConfigs.getPrimary().getProvider()
        );
        
        // 使用 StreamingChatLanguageModel 实现流式输出
        StreamingChatLanguageModel streamingModel = getStreamingModel();
        streamingModel.generate(prompt, new StreamingResponseHandler<AiMessage>() {
            @Override
            public void onNext(String token) {
                onToken.accept(token);
            }
            
            @Override
            public void onError(Throwable error) {
                log.error("Stream error", error);
            }
        });
    }
}
```

### 3.4 基础对话服务实现

```java
@Service
@Slf4j
public class ChatService {
    
    @Autowired
    private ModelRoutingService modelRoutingService;
    
    @Autowired
    private ConversationMemoryService memoryService;
    
    /**
     * 基础对话 - 替换现有Mock实现
     */
    public ChatResponse chat(ChatRequest request) {
        String sessionId = request.getSessionId();
        String userMessage = request.getMessage();
        
        // 1. 获取历史对话
        List<ChatMessage> history = memoryService.getHistory(sessionId);
        
        // 2. 构建Prompt
        String prompt = buildPrompt(userMessage, history);
        
        // 3. 调用大模型
        String response = modelRoutingService.generateWithFallback(prompt);
        
        // 4. 保存对话记录
        memoryService.saveMessage(sessionId, "user", userMessage);
        memoryService.saveMessage(sessionId, "assistant", response);
        
        return ChatResponse.builder()
                .reply(response)
                .type("text")
                .sessionId(sessionId)
                .timestamp(LocalDateTime.now())
                .build();
    }
    
    private String buildPrompt(String userMessage, List<ChatMessage> history) {
        StringBuilder prompt = new StringBuilder();
        
        // 系统提示
        prompt.append("你是一个专业的教育AI助手，擅长:\n");
        prompt.append("1. 解答学习中的疑难问题\n");
        prompt.append("2. 推荐合适的学习资源\n");
        prompt.append("3. 制定个性化学习计划\n");
        prompt.append("4. 鼓励和支持学生\n\n");
        
        // 历史对话
        for (ChatMessage msg : history) {
            prompt.append(msg.getRole()).append(": ")
                  .append(msg.getContent()).append("\n");
        }
        
        // 当前消息
        prompt.append("user: ").append(userMessage).append("\n");
        prompt.append("assistant: ");
        
        return prompt.toString();
    }
}
```

### 3.5 对话记忆服务

```java
@Service
@Slf4j
public class ConversationMemoryService {
    
    @Autowired
    private StringRedisTemplate redisTemplate;
    
    @Autowired
    private ConversationRepository conversationRepository;
    
    private static final String REDIS_KEY_PREFIX = "chat:session:";
    private static final Duration REDIS_TTL = Duration.ofHours(24);
    private static final int MAX_HISTORY = 10; // 保留最近10轮对话
    
    /**
     * 获取对话历史
     */
    public List<ChatMessage> getHistory(String sessionId) {
        String key = REDIS_KEY_PREFIX + sessionId;
        
        // 从Redis获取
        List<String> messages = redisTemplate.opsForList().range(key, 0, -1);
        
        if (messages != null && !messages.isEmpty()) {
            return messages.stream()
                    .map(this::deserializeMessage)
                    .collect(Collectors.toList());
        }
        
        // Redis未命中，从数据库加载
        return loadFromDatabase(sessionId);
    }
    
    /**
     * 保存消息
     */
    public void saveMessage(String sessionId, String role, String content) {
        String key = REDIS_KEY_PREFIX + sessionId;
        ChatMessage message = new ChatMessage(role, content, LocalDateTime.now());
        
        // 保存到Redis
        redisTemplate.opsForList().rightPush(key, serializeMessage(message));
        redisTemplate.opsForList().trim(key, -MAX_HISTORY, -1); // 只保留最近N条
        redisTemplate.expire(key, REDIS_TTL);
        
        // 异步保存到数据库
        saveToDatabaseAsync(sessionId, message);
    }
    
    /**
     * 清除对话历史
     */
    public void clearHistory(String sessionId) {
        String key = REDIS_KEY_PREFIX + sessionId;
        redisTemplate.delete(key);
    }
    
    private String serializeMessage(ChatMessage message) {
        try {
            return new ObjectMapper().writeValueAsString(message);
        } catch (JsonProcessingException e) {
            log.error("Failed to serialize message", e);
            return "";
        }
    }
    
    private ChatMessage deserializeMessage(String json) {
        try {
            return new ObjectMapper().readValue(json, ChatMessage.class);
        } catch (JsonProcessingException e) {
            log.error("Failed to deserialize message", e);
            return null;
        }
    }
}
```

```java
            log.error("Chat error", e);
            return Result.error("AI服务暂时不可用，请稍后重试");
        }
    }
    
    /**
     * 流式对话接口（SSE）
     */
    @PostMapping(value = "/chat/stream", produces = MediaType.TEXT_EVENT_STREAM_VALUE)
    public SseEmitter chatStream(@RequestBody ChatRequest request) {
        SseEmitter emitter = new SseEmitter(300_000L); // 5分钟超时
        
        streamingChatService.generateStream(request, new StreamingResponseHandler<>() {
            @Override
            public void onNext(String token) {
                try {
                    emitter.send(SseEmitter.event()
                            .data(token)
                            .id(UUID.randomUUID().toString()));
                } catch (IOException e) {
                    log.error("SSE send error", e);
                }
            }
            
            @Override
            public void onComplete(Response<AiMessage> response) {
                emitter.complete();
            }
            
            @Override
            public void onError(Throwable error) {
                log.error("Stream error", error);
                emitter.completeWithError(error);
            }
        });
        
        return emitter;
    }
    
    /**
     * 清除对话历史
     */
    @PostMapping("/chat/clear")
    public Result<Void> clearChat(@RequestParam String sessionId) {
        chatService.clearHistory(sessionId);
        return Result.success();
    }

        status.put("service", "running");
```

### 4.3 知识库服务实现

```java
@Service
@Slf4j
public class KnowledgeBaseService {
    
    @Autowired
    private EmbeddingStore<TextSegment> embeddingStore;
    
    @Autowired
    private EmbeddingModel embeddingModel;
    
    @Autowired
    private CourseRepository courseRepository;
    
    /**
     * 初始化知识库 - 加载所有课程数据
     */
    @PostConstruct
    public void initialize() {
        log.info("Initializing knowledge base...");
        
        // 检查是否已初始化
        if (isInitialized()) {
            log.info("Knowledge base already initialized");
            return;
        }
        
        // 加载课程数据
        List<Course> courses = courseRepository.findAll();
        for (Course course : courses) {
            indexCourse(course);
        }
        
        log.info("Knowledge base initialized with {} courses", courses.size());
    }
    
    /**
     * 索引单个课程
     */
    public void indexCourse(Course course) {
        try {
            // 1. 构建课程文档
            String content = buildCourseDocument(course);
            
            // 2. 文本分块
            List<TextSegment> segments = splitText(content);
            
            // 3. 生成向量并存储
            for (TextSegment segment : segments) {
                Embedding embedding = embeddingModel.embed(segment).content();
                
                TextSegment metadataSegment = TextSegment.from(segment.text(), 
                    Metadata.from(Map.of(
                        "courseId", course.getId(),
                        "courseName", course.getName(),
                        "category", course.getCategory(),
                        "difficulty", course.getDifficulty(),
                        "source", "course"
                    ))
                );
                
                embeddingStore.add(embedding, metadataSegment);
            }
            
            log.info("Indexed course: {}", course.getName());
        } catch (Exception e) {
            log.error("Failed to index course: {}", course.getName(), e);
        }
    }
    
    /**
     * 检索相关知识
     */
    public List<RetrievedDocument> retrieve(String query, int maxResults) {
        // 1. 向量化查询
        Embedding queryEmbedding = embeddingModel.embed(query).content();
        
        // 2. 向量检索
        List<EmbeddingMatch<TextSegment>> matches = 
            embeddingStore.findRelevant(queryEmbedding, maxResults);
        
        // 3. 重排序（可选）
        matches = rerank(matches, query);
        
        // 4. 转换为文档
        return matches.stream()
                .map(match -> RetrievedDocument.builder()
                        .content(match.embedded().text())
                        .score(match.score())
                        .metadata(match.embedded().metadata().toMap())
                        .build())
                .collect(Collectors.toList());
    }
    
    /**
     * 构建课程文档
     */
    private String buildCourseDocument(Course course) {
        StringBuilder doc = new StringBuilder();
        doc.append("课程名称：").append(course.getName()).append("\n");
        doc.append("课程类别：").append(course.getCategory()).append("\n");
        doc.append("难度等级：").append(course.getDifficulty()).append("\n");
        doc.append("课程描述：").append(course.getDescription()).append("\n");
        doc.append("学习目标：").append(course.getLearningObjectives()).append("\n");
        doc.append(" prerequisites：").append(course.getPrerequisites()).append("\n");
        
        // 添加章节信息
        if (course.getChapters() != null) {
            doc.append("\n课程大纲：\n");
            for (Chapter chapter : course.getChapters()) {
                doc.append("- ").append(chapter.getTitle()).append("\n");
                doc.append("  ").append(chapter.getDescription()).append("\n");
            }
        }
        
        return doc.toString();
    }
    
    /**
     * 文本分块策略
     */
    private List<TextSegment> splitText(String text) {
        DocumentSplitter splitter = DocumentSplitters.recursive(
                500,    // 每块最大字符数
                50,     // 重叠字符数
                new OpenAiTokenizer()  // 使用tokenizer估算
        );
        
        return splitter.split(Document.from(text));
    }
    
    /**
     * 重排序优化
     */
    private List<EmbeddingMatch<TextSegment>> rerank(
            List<EmbeddingMatch<TextSegment>> matches, String query) {
        // 使用Cross-Encoder进行重排序
        // 或者使用简单的关键词匹配加权
        return matches.stream()
                .sorted((a, b) -> {
                    double scoreA = calculateRelevanceScore(a, query);
                    double scoreB = calculateRelevanceScore(b, query);
                    return Double.compare(scoreB, scoreA);
                })
                .collect(Collectors.toList());
    }
    
    private double calculateRelevanceScore(EmbeddingMatch<TextSegment> match, String query) {
        double vectorScore = match.score();
        String content = match.embedded().text().toLowerCase();
        String queryLower = query.toLowerCase();
        
        // 关键词匹配加分
        double keywordBonus = 0;
        if (content.contains(queryLower)) {
            keywordBonus = 0.1;
        }
        
        return vectorScore + keywordBonus;
    }
}
```

```java
@Service
@Slf4j
public class ToolEnabledAgentService {
    
    @Autowired
    private ToolRegistry toolRegistry;
    
    @Autowired
    private ModelRoutingService modelRoutingService;
    
    /**
     * 处理带工具调用的对话
     */
    public AgentResponse chatWithTools(AgentRequest request) {
        String userMessage = request.getMessage();
        
        // 1. 构建系统提示（包含工具描述）
        String systemPrompt = buildSystemPromptWithTools();
        
        // 2. 调用模型，可能返回工具调用请求
        String modelResponse = modelRoutingService.generateWithFallback(
            systemPrompt + "\n\n用户：" + userMessage + "\n助手："
        );
        
        // 3. 检查是否需要工具调用
        if (isToolCallRequest(modelResponse)) {
            ToolCallRequest toolCall = parseToolCall(modelResponse);
            
            // 执行工具
            AgentTool tool = toolRegistry.getTool(toolCall.getToolName());
            ToolExecuteResult result = tool.execute(toolCall.getParameters());
            
            // 将工具结果返回给模型
            String finalResponse = modelRoutingService.generateWithFallback(
                systemPrompt + "\n\n用户：" + userMessage + "\n" +
                "工具调用结果：" + result.getSummary() + "\n" +
                "详细数据：" + JsonUtils.toJson(result.getData()) + "\n" +
                "请基于以上结果回答用户。\n助手："
            );
            
            return AgentResponse.builder()
                    .content(finalResponse)
                    .type("tool_response")
                    .toolUsed(toolCall.getToolName())
                    .build();
        }
        
        return AgentResponse.builder()
                .content(modelResponse)
                .type("direct")
                .build();
    }
    
    private String buildSystemPromptWithTools() {
        StringBuilder prompt = new StringBuilder();
        prompt.append("你是一个智能教育助手。你可以使用以下工具来帮助用户：\n\n");
        prompt.append(toolRegistry.getToolsDescription());
        prompt.append("\n如果需要使用工具，请按以下格式回复：\n");
        prompt.append("TOOL_CALL: {\"tool\": \"工具名\", \"parameters\": {参数}}\n\n");
        prompt.append("如果不需要工具，直接回答即可。\n");
        return prompt.toString();
    }
    
    private boolean isToolCallRequest(String response) {
        return response.trim().startsWith("TOOL_CALL:");
    }
    
    private ToolCallRequest parseToolCall(String response) {
        String json = response.replace("TOOL_CALL:", "").trim();
        return JsonUtils.fromJson(json, ToolCallRequest.class);
    }
```
